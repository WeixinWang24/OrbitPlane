examples/task_log.py

from pathlib import Path

TASK_FILE = Path("examples/tasks.txt")
tasks = []
def add_task(tasks, title):
    tasks.append(title)
    return tasks
def save_tasks(tasks, file_path):
    text = "\n".join(f"- {task}" for task in tasks)
    file_path.write_text(text + "\n", encoding="utf-8")
    return len(tasks)
if __name__ == "__main__":
    add_task(tasks, "read the code")
    add_task(tasks, "run the script")
    count = save_tasks(tasks, TASK_FILE)
    print(f"saved {count} tasks")
